## 最佳实践

### 图表生成标准流程

```python
import matplotlib.pyplot as plt
import matplotlib
import numpy as np

# 设置中文字体支持（必须在其他导入后立即设置）
# macOS: PingFang SC, STHeiti
# Windows: Microsoft YaHei, SimHei
# Linux: WenQuanYi Micro Hei, DejaVu Sans
matplotlib.rcParams['font.sans-serif'] = ['Arial Unicode MS', 'PingFang SC', 'Microsoft YaHei', 'SimHei', 'WenQuanYi Micro Hei']
matplotlib.rcParams['axes.unicode_minus'] = False  # 解决负号显示问题

# Step 1: 数据准备
x = np.linspace(0, 10, 100)
y = np.sin(x)

# Step 2: 图表配置（使用面向对象API）
fig, ax = plt.subplots(figsize=(10, 6))
ax.plot(x, y, linewidth=2, label='正弦函数')
ax.set_xlabel('X轴', fontsize=11)
ax.set_ylabel('Y轴', fontsize=11)
ax.set_title('正弦函数', fontsize=13, fontweight='bold')
ax.legend(fontsize=10)
ax.grid(True, linestyle='--', alpha=0.6)

# Step 3: 高质量保存
fig.savefig(
    'sine_function.png',
    dpi=300,           # 高分辨率
    bbox_inches='tight',  # 避免标签裁剪
    facecolor='white',
    edgecolor='none'
)
plt.close(fig)  # 释放内存

print('图表已保存: sine_function.png (300 DPI)')
```
