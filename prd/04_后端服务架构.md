### 3.2 业务服务层

#### 3.2.1 Loop队列

**文件**: `chatbi/core/loop_queue.py`

```python
import asyncio
from typing import Callable, Optional
from loguru import logger
from ..models import Message
from .agent_wrapper import AgentWrapper

class LoopQueue:
    _instance: Optional["LoopQueue"] = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance
    
    def __init__(self):
        if hasattr(self, '_initialized'):
            return
        self._initialized = True
        self.queue = asyncio.Queue()
        self.agent_wrapper = AgentWrapper()
        self._running = False
        self._workers = []
    
    @classmethod
    def get_instance(cls) -> "LoopQueue":
        if cls._instance is None:
            cls._instance = cls()
        return cls._instance
    
    async def start(self, num_workers: int = 4):
        """启动worker池"""
        self._running = True
        for i in range(num_workers):
            worker = asyncio.create_task(self._worker(f"worker-{i}"))
            self._workers.append(worker)
        logger.info(f"Loop队列已启动，{num_workers}个worker")
    
    async def stop(self):
        """停止worker池"""
        self._running = False
        for worker in self._workers:
            worker.cancel()
        await asyncio.gather(*self._workers, return_exceptions=True)
        logger.info("Loop队列已停止")
    
    async def enqueue(self, message: Message):
        """入队消息"""
        await self.queue.put(message)
        logger.debug(f"消息入队: {message.id}")
    
    async def _worker(self, name: str):
        """Worker处理消息"""
        logger.info(f"{name} 启动")
        while self._running:
            try:
                message = await asyncio.wait_for(self.queue.get(), timeout=1.0)
                logger.info(f"{name} 处理消息: {message.id}")
                await self.agent_wrapper.process(message)
                self.queue.task_done()
            except asyncio.TimeoutError:
                continue
            except Exception as e:
                logger.error(f"{name} 处理消息失败: {e}")
```

#### 3.2.2 Agent包装器

**文件**: `chatbi/core/agent_wrapper.py`

```python
from pathlib import Path
from typing import Optional
from loguru import logger
from nanobot.agent.loop import AgentLoop
from nanobot.provider.litellm_provider import LiteLLMProvider
from nanobot.session.manager import SessionManager
from ..models import Message, Conversation
from .conversation_manager import ConversationManager
from ..agent.tools.registry import ToolRegistry

class AgentWrapper:
    def __init__(self):
        self.loop: Optional[AgentLoop] = None
        self.conversation_manager = ConversationManager()
        self._initialized = False
    
    async def initialize(self):
        """初始化Agent Loop"""
        if self._initialized:
            return
        
        provider = LiteLLMProvider(model="gpt-4")
        workspace = Path("/workspace")
        session_manager = SessionManager(workspace)
        
        self.loop = AgentLoop(
            bus=MessageBus(),
            provider=provider,
            workspace=workspace,
            session_manager=session_manager,
            max_iterations=20,
        )
        
        # 注册ChatBI工具
        self._register_tools()
        
        self._initialized = True
        logger.info("Agent包装器初始化完成")
    
    def _register_tools(self):
        """注册ChatBI工具"""
        from ..agent.tools.file_ops import ReadFileTool, WriteFileTool
        from ..agent.tools.rag_tool import RAGTool
        from ..agent.tools.sql_tool import SQLTool
        from ..agent.tools.python_tool import PythonTool
        
        self.loop.tools.register(ReadFileTool())
        self.loop.tools.register(WriteFileTool())
        self.loop.tools.register(RAGTool())
        self.loop.tools.register(SQLTool())
        self.loop.tools.register(PythonTool())
    
    async def process(self, message: Message):
        """处理消息"""
        await self.initialize()
        
        conversation = self.conversation_manager.get(message.conversation_id)
        if not conversation:
            logger.error(f"对话不存在: {message.conversation_id}")
            return
        
        # 创建InboundMessage
        from nanobot.bus.events import InboundMessage
        inbound = InboundMessage(
            channel=message.channel,
            sender_id=message.user_id,
            chat_id=message.conversation_id,
            content=message.content,
            metadata={"scene_code": conversation.scene_code}
        )
        
        # 处理消息
        response = await self.loop._process_message(inbound)
        
        # 保存响应
        self.conversation_manager.add_message(
            message.conversation_id,
            role="assistant",
            content=response.content if response else "",
        )
```

import httpx
from typing import Optional, List, Dict, Any
from ..base import BaseTool, tool_result
from loguru import logger

class RAGTool(BaseTool):
    name = "rag_search"
    description = "查询RAG知识库获取文档、表结构、QA示例"
    
    parameters = {
        "type": "object",
        "properties": {
            "scene_code": {"type": "string", "description": "场景编码"},
            "query": {"type": "string", "description": "用户问题"},
            "type": {"type": "string", "enum": ["doc", "schema", "qa", "all"], "default": "all"}
        },
        "required": ["scene_code", "query"]
    }
    
    async def execute(self, scene_code: str, query: str, type: str = "all") -> Dict[str, Any]:
        """执行RAG查询"""
        try:
            async with httpx.AsyncClient(timeout=30.0) as client:
                response = await client.post(
                    "http://my_rag_v1",
                    json={
                        "scene_code": scene_code,
                        "query": query,
                        "type": type
                    }
                )
                response.raise_for_status()
                data = response.json()
                
                # 格式化结果
                result = {
                    "success": True,
                    "docs": data.get("docs", []),
                    "schemas": data.get("schemas", []),
                    "qa_examples": data.get("qa_examples", [])
                }
                
                logger.info(f"RAG查询成功: {len(result['docs'])} docs, {len(result['schemas'])} schemas")
                return tool_result(result)
                
        except Exception as e:
            logger.error(f"RAG查询失败: {e}")
            return tool_result(f"RAG查询失败: {str(e)}", success=False)

import httpx
from typing import Dict, Any
from ..base import BaseTool, tool_result
from loguru import logger

class SQLTool(BaseTool):
    name = "execute_sql"
    description = "执行SQL查询语句（仅支持SELECT）"
    
    parameters = {
        "type": "object",
        "properties": {
            "sql_text": {"type": "string", "description": "SQL语句"},
            "scene_code": {"type": "string", "description": "场景编码"}
        },
        "required": ["sql_text", "scene_code"]
    }
    
    async def execute(self, sql_text: str, scene_code: str) -> Dict[str, Any]:
        """执行SQL查询"""
        # 安全检查：仅允许SELECT
        sql_upper = sql_text.upper().strip()
        if not sql_upper.startswith("SELECT"):
            return tool_result("仅支持SELECT查询语句", success=False)
        
        # 自动添加LIMIT
        if "LIMIT" not in sql_upper:
            sql_text += " LIMIT 10000"
        
        try:
            async with httpx.AsyncClient(timeout=30.0) as client:
                response = await client.post(
                    "http://my_hive_sql_exec",
                    json={
                        "sql_text": sql_text,
                        "scene_code": scene_code
                    }
                )
                response.raise_for_status()
                data = response.json()
                
                logger.info(f"SQL执行成功: {data.get('row_count')} 行")
                return tool_result({
                    "success": True,
                    "data": data.get("data", []),
                    "columns": data.get("columns", []),
                    "row_count": data.get("row_count", 0),
                    "execution_time_ms": data.get("execution_time_ms", 0)
                })
                
        except Exception as e:
            logger.error(f"SQL执行失败: {e}")
            return tool_result(f"SQL执行失败: {str(e)}", success=False)

from pathlib import Path
from typing import Dict, Any
from opensandbox import Sandbox
from ..base import BaseTool, tool_result
from loguru import logger

class PythonTool(BaseTool):
    name = "execute_python"
    description = "在沙箱中执行Python代码"
    
    parameters = {
        "type": "object",
        "properties": {
            "code": {"type": "string", "description": "Python代码"},
            "timeout": {"type": "integer", "default": 60, "description": "超时时间（秒）"}
        },
        "required": ["code"]
    }
    
    async def execute(self, code: str, timeout: int = 60) -> Dict[str, Any]:
        """执行Python代码"""
        try:
            # 创建沙箱
            sandbox = Sandbox(
                cpu_limit=1,
                memory_limit="512MB",
                disk_limit="1GB",
                network=False,  # 禁用网络
                timeout=timeout
            )
            
            # 执行代码
            result = sandbox.execute(code)
            
            if result.success:
                logger.info("Python执行成功")
                return tool_result({
                    "success": True,
                    "output": result.stdout,
                    "error": result.stderr
                })
            else:
                logger.error(f"Python执行失败: {result.stderr}")
                return tool_result(f"Python执行失败: {result.stderr}", success=False)
                
        except Exception as e:
            logger.error(f"Python沙箱执行失败: {e}")
            return tool_result(f"Python执行失败: {str(e)}", success=False)

#### 3.3.4 文件操作工具

**文件**: `chatbi/agent/tools/file_ops.py`

```python
from pathlib import Path
from typing import Dict, Any, Optional
from ..base import BaseTool, tool_result
from loguru import logger

class ReadFileTool(BaseTool):
    name = "read_file"
    description = "读取文件内容"
    
    parameters = {
        "type": "object",
        "properties": {
            "file_path": {"type": "string", "description": "文件路径（相对路径）"},
            "limit": {"type": "integer", "default": 100, "description": "读取行数限制"}
        },
        "required": ["file_path"]
    }
    
    async def execute(self, file_path: str, limit: int = 100, user_channel: str = "") -> Dict[str, Any]:
        """读取文件"""
        try:
            full_path = Path(f"/workspace/files/{user_channel}/{file_path}")
            if not full_path.exists():
                return tool_result(f"文件不存在: {file_path}", success=False)
            
            lines = []
            with open(full_path, 'r', encoding='utf-8') as f:
                for i, line in enumerate(f):
                    if i >= limit:
                        break
                    lines.append(line.rstrip('\n'))
            
            content = '\n'.join(lines)
            logger.info(f"读取文件成功: {file_path} ({len(lines)} 行)")
            return tool_result({"success": True, "content": content})
            
        except Exception as e:
            logger.error(f"读取文件失败: {e}")
            return tool_result(f"读取文件失败: {str(e)}", success=False)

class WriteFileTool(BaseTool):
    name = "write_file"
    description = "写入文件内容"
    
    parameters = {
        "type": "object",
        "properties": {
            "file_path": {"type": "string", "description": "文件路径（相对路径）"},
            "content": {"type": "string", "description": "文件内容"},
            "description": {"type": "string", "description": "文件说明"}
        },
        "required": ["file_path", "content"]
    }
    
    async def execute(self, file_path: str, content: str, description: str = "", user_channel: str = "") -> Dict[str, Any]:
        """写入文件"""
        try:
            full_path = Path(f"/workspace/files/{user_channel}/{file_path}")
            full_path.parent.mkdir(parents=True, exist_ok=True)
            
            with open(full_path, 'w', encoding='utf-8') as f:
                f.write(content)
            
            logger.info(f"写入文件成功: {file_path}")
            return tool_result({"success": True, "file_path": str(full_path)})
            
        except Exception as e:
            logger.error(f"写入文件失败: {e}")
            return tool_result(f"写入文件失败: {str(e)}", success=False)
```

### 3.5 数据模型层

**文件**: `chatbi/models/conversation.py`

from pydantic import BaseModel, Field
from typing import List, Optional
from datetime import datetime
from enum import Enum

class ConversationStatus(str, Enum):
    ACTIVE = "active"
    ARCHIVED = "archived"
    DELETED = "deleted"

class Message(BaseModel):
    role: str = Field(..., description="角色: user/assistant")
    content: str = Field(..., description="消息内容")
    timestamp: datetime = Field(default_factory=datetime.utcnow, description="时间戳")
    tools_used: Optional[List[str]] = None, Field(description="使用的工具")
    metadata: Optional[dict] = Field(default_factory=dict)

class Conversation(BaseModel):
    conversation_id: str = Field(..., description="对话ID")
    user_channel: str = Field(..., description="用户Channel标识")
    scene_code: str = Field(..., description="场景编码")
    scene_name: str = Field(..., description="场景名称")
    created_at: datetime = Field(default_factory=datetime.utcnow, description="创建时间")
    updated_at: datetime = Field(default_factory=datetime.utcnow, description="更新时间")
    status: ConversationStatus = Field(default=ConversationStatus.ACTIVE, description="状态")
    messages: List[Message] = Field(default_factory=list, description="消息列表")
